always @(posedge clk or posedge reset) begin
    if (reset) begin
        present_state <= 0;
    end else begin
        present_state <= next_state;
    end
end

always @* begin
    next_state = present_state;
    case (present_state)
        0: begin
            out = 1;
            if (in == 0) begin
                next_state = 1;
            end
        end
        1: begin
            out = 0;
            if (in == 0) begin
                next_state = 0;
            end
        end
    endcase
end
endmodule
