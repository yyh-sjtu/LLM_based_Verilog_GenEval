always @(posedge clk or posedge reset) begin
    if(reset) begin
        present_state <= 0;
    end else begin
        present_state <= next_state;
    end
end

always @* begin
    case(present_state)
        0: begin
            out = 1;
            if(in == 0)
                next_state = 1;
            else
                next_state = 0;
        end
        1: begin
            out = 0;
            if(in == 0)
                next_state = 0;
            else
                next_state = 1;
        end
    endcase
end
endmodule
