// Define initial state of LFSR
initial begin
    r_reg <= 5'b1; // Initial seed value
end

// Calculate next state based on taps
assign feedback_value = r_reg[4] ^ r_reg[2] ^ r_reg[0]; // XOR taps at bits 5 and 3
assign r_next = {feedback_value, r_reg[4:1]}; // Shift feedback value to msb position

always @(posedge clk or posedge reset) begin
    if (reset) begin
        r_reg <= 5'b1; // Reset to initial seed value
    end else begin
        r_reg <= r_next; // Update state based on clock
    end
end

// Output current state
assign q = r_reg;

endmodule
