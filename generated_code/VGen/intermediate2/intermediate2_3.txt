module counter( 
  input clk,
  input reset,
  output reg [3:0] q
);

  always @(posedge clk) begin
    if(reset) begin
      q <= 4'b0001; // reset to 1
    end else begin
      if(q == 4'b1100) begin
        q <= 4'b0001; // reset to 1 when q reaches 12
      end else begin
        q <= q + 1; // increment q by 1
      end
    end
  end
endmodule
