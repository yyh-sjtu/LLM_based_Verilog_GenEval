// Verilog code for implementing an AND gate
module and_gate(
    input a,     // Input 'a' to AND gate
    input b,     // Input 'b' to AND gate
    output out   // Output of AND gate
);

    assign out = a & b;   // AND operation on input 'a' and 'b'

endmodule
