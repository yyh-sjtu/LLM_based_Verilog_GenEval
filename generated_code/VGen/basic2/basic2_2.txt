module full_adder(
    input a,
    input b,
    input cin,
    output sum,
    output cout
    );

    assign {cout, sum} = a + b + cin;

endmodule

// Testbench for the full adder module

module testbench;
    reg a, b, cin;
    wire sum, cout;

    full_adder dut(
        .a(a),
        .b(b),

        .sum(sum),
        .cout(cout)
    );

    initial begin

endmodule
